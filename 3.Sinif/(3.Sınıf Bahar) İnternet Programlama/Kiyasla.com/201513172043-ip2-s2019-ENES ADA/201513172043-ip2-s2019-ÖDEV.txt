import javax.swing.JOptionPane;
public class Zar {

int bir = 0;
int iki = 0;
int uc = 0;

int alti = 0;
int rs = 0;

case 1:

case 2:

case 3:
uc++;
break;
case 4:
dort++;
break;
case 5:
bes++;

case 6:

}
}

}
